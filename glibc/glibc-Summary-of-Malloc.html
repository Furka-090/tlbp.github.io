3.2.3.10. malloc ve İlgili İşlevlerin Özeti

Burada, malloc ile çalışan işlevlerin bir özeti bulunmaktadır:

      işlev
#include <stdlib.h>
void *malloc(size_t  size)

size baytlık bellek dilimi tahsis eder. Bkz. Özdevimli Olarak Basit Bellek Tahsisi.

      işlev
#include <stdlib.h>
void  free(void *addr)

Evvelce, malloc ile tahsis edilmiş belleği serbest bırakır. Bkz. malloc ile Ayrılan Belleğin Serbest Bırakılması.

      işlev
#include <stdlib.h>
void *realloc(void *addr, size_t  size)

Evvelce malloc ile tahsis edilmiş bellek dilimini muhtemelen başka bir yere kopyalamak suretiyle büyütür veya küçültür. Bkz. Bellek Diliminin Boyutunun Değiştirilmesi.

      işlev
#include <stdlib.h>
void *reallocarray(void *ptr, size_t  nmemb, size_t  size)

Evvelce malloc ile tahsis edilmiş bellek diliminin boyunu realloc işlevine benzer şekilde nmemb * size bayta değiştirir. Bkz. Bellek Diliminin Boyutunun Değiştirilmesi.

      işlev
#include <stdlib.h>
void *calloc(size_t  count, size_t  eltsize)

malloc işlevini kullanarak count * eltsize baytlık bellek dilimi ayırır. Bkz. Temizlenmiş Bellek Tahsisi.

      işlev
#include <stdlib.h>
void *valloc(size_t  size)

Sayfa kenarından başlayan size baytlık bellek dilimi ayırır. Bkz. Bellek Dilimlerinin Hizalanarak Tahsisi.

      işlev
#include <stdlib.h>
void *aligned_alloc(size_t  size, size_t  alignment)

alignment değerinin katlarındaki bir adresten başlayan size baytlık bellek dilimi ayırır. Bkz. Bellek Dilimlerinin Hizalanarak Tahsisi.

      işlev
#include <stdlib.h>
int  posix_memalign(void **memptr, size_t  alignment, size_t  size)

alignment değerinin katlarındaki bir adresten başlayan size baytlık bellek dilimi ayırır. Bkz. Bellek Dilimlerinin Hizalanarak Tahsisi.

      işlev
#include <stdlib.h>
void *memalign(size_t  size, size_t  boundary)

boundary değerinin katlarındaki bir adresten başlayan size baytlık bellek dilimi ayırır. Bkz. Bellek Dilimlerinin Hizalanarak Tahsisi.

      işlev
#include <malloc.h>
int  mallopt(int  param, int  value)

Ayarlanabilir bağımsız değişkenleri ayarlar. Bkz. malloc Ayarlanabilirleri.

      işlev
#include <mcheck.h>
int  mcheck(void(*abortfn) (enum mcheck_status status))

malloc işlevine özdevimli tahsis edilmiş dilim üzerinde arasıra tutarlılık denetimi yapmasını ve bir tutarsızlık bulunursa abortfn işlevini çağırmasını söyler. Bkz. Yığın Bellek Tutarlılık Denetimi.

      işlev
#include <malloc.h>
struct mallinfo2  mallinfo2(void )

Mevcut özdevimli bellek kullanımı hakkında bilgi döndürür. Bkz. malloc ile Bellek Tahsis İstatistikleri.
